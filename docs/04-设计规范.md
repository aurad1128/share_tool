### 阴影规范（QGraphicsDropShadowEffect）

```python
from PyQt5.QtWidgets import QGraphicsDropShadowEffect
from PyQt5.QtGui import QColor
from PyQt5.QtCore import QPointF

# 创建阴影效果
def create_shadow(blur_radius=8, offset=(0, 2), color='#FFB6C1', opacity=0.3):
    shadow = QGraphicsDropShadowEffect()
    shadow.setBlurRadius(blur_radius)
    shadow.setOffset(QPointF(offset[0], offset[1]))
    shadow_color = QColor(color)
    shadow_color.setAlphaF(opacity)
    shadow.setColor(shadow_color)
    return shadow

# 使用示例
button.setGraphicsEffect(create_shadow(blur_radius=8, offset=(0, 2)))
```

```python
WINDOW_CONFIG = {
    'default_width': 800,
    'default_height': 600,
    'min_width': 600,
    'min_height': 400,
}

# 在主窗口中使用
class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("面对面传输工具")
        self.resize(WINDOW_CONFIG['default_width'], 
                   WINDOW_CONFIG['default_height'])
        self.setMinimumSize(WINDOW_CONFIG['min_width'], 
                           WINDOW_CONFIG['min_height'])
        # 窗口居中
        self.center_window()
    
    def center_window(self):
        screen = QApplication.desktop().screenGeometry()
        size = self.geometry()
        self.move((screen.width() - size.width()) // 2,
                 (screen.height() - size.height()) // 2)
```

```python
class LoadingWidget(QLabel):

            }

    def update_loading(self):
        self.dots = (self.dots + 1) % 4
        self.setText("加载中" + "." * self.dots)
```

```python
from PyQt5.QtCore import QParallelAnimationGroup

def switch_page_with_animation(stacked_widget, index):
    # 淡出当前页面
    current_widget = stacked_widget.currentWidget()
    fade_out = QPropertyAnimation(current_widget, b"windowOpacity")
    fade_out.setDuration(150)
    fade_out.setStartValue(1)
    fade_out.setEndValue(0)
    
    # 切换页面
    def on_fade_out_finished():
        stacked_widget.setCurrentIndex(index)
        # 淡入新页面
        new_widget = stacked_widget.currentWidget()
        fade_in = QPropertyAnimation(new_widget, b"windowOpacity")
        fade_in.setDuration(150)
        fade_in.setStartValue(0)
        fade_in.setEndValue(1)
        fade_in.start()
    
    fade_out.finished.connect(on_fade_out_finished)
    fade_out.start()
```

## 响应式设计

### 窗口大小适配
```python
class ResponsiveWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.init_ui()
    
    def init_ui(self):
        self.layout = QVBoxLayout()
        self.setLayout(self.layout)
    
    def resizeEvent(self, event):
        """窗口大小改变时调用"""
        width = event.size().width()
        
        # 根据窗口宽度调整布局
        if width > 800:
            # 大窗口布局
            self.layout.setContentsMargins(48, 32, 48, 32)
        elif width > 600:
            # 中窗口布局
            self.layout.setContentsMargins(32, 24, 32, 24)
        else:
            # 小窗口布局
            self.layout.setContentsMargins(16, 16, 16, 16)
        
        super().resizeEvent(event)
```

```python
class AdaptiveButton(QPushButton):
    def __init__(self, text):
        super().__init__(text)
        self.setMinimumWidth(200)
    
    def resizeEvent(self, event):
        parent_width = self.parent().width() if self.parent() else 800
        if parent_width < 600:
            self.setMaximumWidth(parent_width - 32)
        else:
            self.setMaximumWidth(400)
        super().resizeEvent(event)
```

### 提示文案原则
- **简洁明了** - 避免技术术语
- **提供解决方案** - 告诉用户怎么做
- **友好语气** - 不要责怪用户

**示例：**
- ❌ 不好："Error: Connection timeout (code: 504)"
- ✅ 好："网络连接超时，请检查网络后重试"

- ❌ 不好："Invalid input format"
- ✅ 好："暗号格式不正确，请输入6-8位字母或数字"
